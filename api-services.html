JavaScript Integration Example

// Initialize Trident Systems API Client const TridentAPI = { baseURL: 'https://api.tridentsys.ca/v1', apiKey: 'your_api_key_here', async getHydrants(bounds) { const response = await fetch(`${this.baseURL}/hydrants?bounds=${bounds}`, { headers: { 'Authorization': `Bearer ${this.apiKey}`, 'Content-Type': 'application/json' } }); return await response.json(); }, async getFlowTests(hydrantId) { const response = await fetch(`${this.baseURL}/hydrants/${hydrantId}/flow-tests`, { headers: { 'Authorization': `Bearer ${this.apiKey}`, 'Content-Type': 'application/json' } }); return await response.json(); } }; // Usage Example TridentAPI.getHydrants('43.51,-79.90,43.53,-79.86') .then(data => { console.log(`Found ${data.total} hydrants`); data.data.forEach(hydrant => { console.log(`Hydrant ${hydrant.id}: ${hydrant.status}`); }); }) .catch(error => console.error('API Error:', error));
