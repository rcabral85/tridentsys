// JavaScript SDK Example
const TridentAPI = require('@tridentsys/hydrant-api');

const client = new TridentAPI({
  apiKey: process.env.TRIDENT_API_KEY,
  baseURL: 'https://api.tridentsys.ca/v1'
});

// Get nearby hydrants for emergency dispatch
async function getNearbyHydrants(latitude, longitude) {
  try {
    const hydrants = await client.hydrants.search({
      lat: latitude,
      lon: longitude,
      radius: 500, // meters
      status: 'active'
    });
    
    return hydrants.map(h => ({
      id: h.id,
      distance: h.distance,
      flowCapacity: h.latest_flow_test.available_flow_20psi,
      lastTested: h.latest_flow_test.test_date,
      coordinates: [h.longitude, h.latitude]
    }));
  } catch (error) {
    console.error('API Error:', error.message);
    throw error;
  }
}

// Real-time status monitoring

# Python SDK Example
import requests
from datetime import datetime

class TridentHydrantAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = 'https://api.tridentsys.ca/v1'
        self.session = requests.Session()
        self.session.headers.update({
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        })
    
    def get_hydrant_status(self, hydrant_id):
        """Get comprehensive status for a specific hydrant"""
        response = self.session.get(f'{self.base_url}/hydrants/{hydrant_id}')
        response.raise_for_status()
        return response.json()
    
    def get_maintenance_due(self, municipality_id, days_ahead=30):
        """Get hydrants requiring maintenance in next N days"""
        params = {
            'municipality_id': municipality_id,
            'maintenance_due_within_days': days_ahead,
            'status': 'active'
        }
        response = self.session.get(f'{self.base_url}/hydrants/maintenance-due', params=params)
        return response.json()
    
    def submit_flow_test(self, hydrant_id, test_data):
        """Submit new NFPA 291 flow test results"""
        payload = {
            'test_date': datetime.now().isoformat(),
            'operator_license': test_data['operator_license'],
            'static_pressure': test_data['static_pressure'],
            'residual_pressure': test_data['residual_pressure'],
            'flow_rate': test_data['flow_rate'],
            'nfpa_compliant': True
        }
        response = self.session.post(f'{self.base_url}/hydrants/{hydrant_id}/flow-tests', json=payload)
        return response.json()
